import pandas as pd import seaborn as sns import matplotlib.pyplot as plt data = pd.read_csv("housing.csv") filtered = data.select_dtypes(include=['int64', 'float64']) core = filtered.corr() plt.figure(figsize=(10,8)) sns.heatmap(core, annot=True, cmap='coolwarm', fmt='.2f', linewidth=0.5) plt.title("Housing Data Analysis") plt.show() sns.pairplot(filtered) plt.show()
